document.getElementById("search").addEventListener("keyup",function(e){ let value=this.value.toLowerCase() let gurus=document.querySelectorAll(".guru") gurus.forEach(g=>{ g.classList.remove("highlight") if(g.dataset.name.includes(value)){ g.style.display="flex" g.classList.add("highlight") } else{ g.style.display="none" g.nextElementSibling.style.maxHeight=null } }) })
